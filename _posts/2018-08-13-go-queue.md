## Queue

A Queue is a linear structure which follows a particular order in which the operations are performed. The order is First In First Out (FIFO).

```java

package main

import (
	"fmt"
)

type node struct {
	next *node
	data int
}

type queue struct {
	head *node
	tail *node
}

func (q *queue) empty() bool {
	return q.head == nil
}

func (q *queue) peek() int {
	return q.head.data
}

func (q *queue) add(data int) {
	n := new(node)
	n.data = data
	if q.tail != nil {
		q.tail.next = n
	}
	q.tail = n
	if q.head == nil {
		q.head = n
	}
}

func (q *queue) remove() {
	q.head = q.head.next
	if q.head == nil {
		q.tail = nil
	}
}

func main() {

	fmt.Println("Creating Queue")

	q := &queue{}
	q.add(10)
	q.add(2)
	q.add(5)

	q.remove()
	fmt.Println(q.peek())
}
```
