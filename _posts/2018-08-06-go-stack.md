## Stack

Stack is a data structure which follows a particular order in which the operations are performed. The order may be LIFO(Last In First Out) or FILO(First In Last Out).

```bash

package main

import (
	"fmt"
)

type node struct {
	next *node
	data int
}

type stack struct {
	top *node
}

func (s *stack) empty() bool {
	return s.top == nil
}

func (s *stack) peek() int {
	return s.top.data
}

func (s *stack) add(data int) {
	n := new(node)
	n.data = data
	n.next = s.top
	s.top = n
}

func (s *stack) pop() {
	s.top = s.top.next
}

func main() {

	fmt.Println("Creating Stack")

	s := &stack{}
	s.add(1)
	s.add(2)
	s.add(3)

	s.pop()
	fmt.Println(s.peek())
}
```
