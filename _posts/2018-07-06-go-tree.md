## Tree Data Structure

A tree data structure can be defined recursively as a collection of nodes (starting at a root node), where each node is a data structure consisting of a value, together with a list of references to nodes (the "children"), with the constraints that no reference is duplicated, and none points to the root.

```bash

package main

import (
	"fmt"
)

type node struct {
	left  *node
	right *node
	data  int
}

func (n *node) insert(value int) {

	if value <= n.data {

		if n.left == nil {
			n.left = &node{}
			n.left.data = value
		} else {
			n.left.insert(value)
		}

	} else {

		if n.right == nil {
			n.right = &node{}
			n.right.data = value
		} else {
			n.right.insert(value)
		}

	}
}

func (n node) contains(value int) bool {

	if value == n.data {
		return true
	} else if value < n.data {

		if n.left == nil {
			return false
		}
		return n.left.contains(value)

	} else {

		if n.right == nil {
			return false
		}

		return n.right.contains(value)
	}
}

func (n node) printInOrder() {
	if n.left != nil {
		n.left.printInOrder()
	}
	fmt.Println(n.data)
	if n.right != nil {
		n.right.printInOrder()
	}
}

// Test Example
func main() {
	fmt.Println("Creating Tree")
	n := &node{}
	n.insert(1)
	n.insert(1)
	n.insert(2)
	n.insert(3)
	n.insert(5)
	n.insert(8)
	fmt.Println(n.contains(1))
	n.printInOrder()
}
```
